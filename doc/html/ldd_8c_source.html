278 static int check_elf_header(ElfW(Ehdr) *const ehdr)
279 {
280  if (!ehdr || *(uint32_t*)ehdr != ELFMAG_U32
281  || ehdr->e_ident[EI_CLASS] != ELFCLASSM
282  || ehdr->e_ident[EI_VERSION] != EV_CURRENT
283  ) {
284  return 1;
285  }
286 
287  /* Check if the target endianness matches the host's endianness */
288  byteswap = 0;
289  if ( __BYTE_ORDER == __LITTLE_ENDIAN) {
290  if (ehdr->e_ident[5] == ELFDATA2MSB)
291  byteswap = 1;
292  } else if ( __BYTE_ORDER == __BIG_ENDIAN) {
293  if (ehdr->e_ident[5] == ELFDATA2LSB)
294  byteswap = 1;
295  }
296 
297  /* Be very lazy, and only byteswap the stuff we use */
298  if (byteswap) {
299  ehdr->e_type = bswap_16(ehdr->e_type);
300  ehdr->e_phoff = byteswap_to_host(ehdr->e_phoff);
301  ehdr->e_shoff = byteswap_to_host(ehdr->e_shoff);
302  ehdr->e_phnum = bswap_16(ehdr->e_phnum);
303  ehdr->e_shnum = bswap_16(ehdr->e_shnum);
304  }
305 
306  return 0;
307 }

316 /* This function's behavior must exactly match that
317  * in uClibc/ldso/ldso/dl-elf.c */
318 static void search_for_named_library(char *name, char *result,
319  const char *path_list)
320 {
321  int i, count = 1;
322  char *path, *path_n;
323  struct stat filestat;
324 
325  /* We need a writable copy of this string */
326  path = strdup(path_list);
327  if (!path) {
328  fprintf(stderr, "%s: Out of memory!\n", __func__);
329  exit(EXIT_FAILURE);
330  }
331  /* Eliminate all double //s */
332  path_n = path;
333  while ((path_n = strstr(path_n, "//"))) {
334  i = strlen(path_n);
335  memmove(path_n, path_n + 1, i - 1);
336  *(path_n + i - 1) = '\0';
337  }
338 
339  /* Replace colons with zeros in path_list and count them */
340  for (i = strlen(path); i > 0; i--) {
341  if (path[i] == ':') {
342  path[i] = 0;
343  count++;
344  }
345  }
346  path_n = path;
347  for (i = 0; i < count; i++) {
348  strcpy(result, path_n);
349  strcat(result, "/");
350  strcat(result, name);
351  if (stat(result, &filestat) == 0 && filestat.st_mode & S_IRUSR) {
352  free(path);
353  return;
354  }
355  path_n += (strlen(path_n) + 1);
356  }
357  free(path);
358  *result = '\0';
359 }

361 static void locate_library_file(ElfW(Ehdr) *ehdr, ElfW(Dyn) *dynamic,
362  int is_suid, struct library *lib)
363 {
364  char *buf;
365  char *path;
366  struct stat filestat;
367 
368  /* If this is a fully resolved name, our job is easy */
369  if (stat(lib->name, &filestat) == 0) {
370  lib->path = strdup(lib->name);
371  return;
372  }
373 
374  /* We need some elbow room here. Make some room... */
375  buf = malloc(1024);
376  if (!buf) {
377  fprintf(stderr, "%s: Out of memory!\n", __func__);
378  exit(EXIT_FAILURE);
379  }
380 
381  /* This function must match the behavior of _dl_load_shared_library
382  * in readelflib1.c or things won't work out as expected... */
383 
384  /* The ABI specifies that RPATH is searched first, so do that now. */
385  path = elf_find_rpath(ehdr, dynamic);
386  if (path) {
387  search_for_named_library(lib->name, buf, path);
388  if (*buf != '\0') {
389  lib->path = buf;
390  return;
391  }
392  }
393 
394  /* Next check LD_{ELF_}LIBRARY_PATH if specified and allowed.
395  * Since this app doesn't actually run an executable I will skip
396  * the suid check, and just use LD_{ELF_}LIBRARY_PATH if set */
397  if (is_suid == 1)
398  path = NULL;
399  else
400  path = getenv("LD_LIBRARY_PATH");
401  if (path) {
402  search_for_named_library(lib->name, buf, path);
403  if (*buf != '\0') {
404  lib->path = buf;
405  return;
406  }
407  }
408 
409  /* Next look for libraries wherever the shared library
410  * loader was installed -- this is usually where we
411  * should find things... */
412  if (interp_dir) {
413  search_for_named_library(lib->name, buf, interp_dir);
414  if (*buf != '\0') {
415  lib->path = buf;
416  return;
417  }
418  }
419 
420  /* Lastly, search the standard list of paths for the library.
421  This list must exactly match the list in uClibc/ldso/ldso/dl-elf.c */
422  path = "/lib:/usr/lib:/usr/X11R6/lib"
423  ;
424  search_for_named_library(lib->name, buf, path);
425  if (*buf != '\0') {
426  lib->path = buf;
427  } else {
428  free(buf);
429  lib->path = not_found;
430  }
431 }

433 static int add_library(ElfW(Ehdr) *ehdr, ElfW(Dyn) *dynamic, int is_setuid, char *s)
434 {
435  char *tmp, *tmp1, *tmp2;
436  struct library *cur, *newlib = lib_list;
437 
438  if (!s || !strlen(s))
439  return 1;
440 
441  tmp = s;
442  while (*tmp) {
443  if (*tmp == '/')
444  s = tmp + 1;
445  tmp++;
446  }
447 
448  /* We add ldso elsewhere */
449  if (interpreter_already_found && (tmp = strrchr(interp_name, '/')) != NULL) {
450  int len = strlen(interp_dir);
451  if (strcmp(s, interp_name + 1 + len) == 0)
452  return 1;
453  }
454 
455  for (cur = lib_list; cur; cur = cur->next) {
456  /* Check if this library is already in the list */
457  tmp1 = tmp2 = cur->name;
458  while (*tmp1) {
459  if (*tmp1 == '/')
460  tmp2 = tmp1 + 1;
461  tmp1++;
462  }
463  if (strcmp(tmp2, s) == 0) {
464  /*printf("find_elf_interpreter is skipping '%s' (already in list)\n", cur->name); */
465  return 0;
466  }
467  }
468 
469  /* Ok, this lib needs to be added to the list */
470  newlib = malloc(sizeof(struct library));
471  if (!newlib)
472  return 1;
473  newlib->name = malloc(strlen(s) + 1);
474  strcpy(newlib->name, s);
475  newlib->resolved = 0;
476  newlib->path = NULL;
477  newlib->next = NULL;
478 
479  /* Now try and locate where this library might be living... */
480  locate_library_file(ehdr, dynamic, is_setuid, newlib);
481 
482  /*printf("add_library is adding '%s' to '%s'\n", newlib->name, newlib->path); */
483  if (!lib_list) {
484  lib_list = newlib;
485  } else {
486  for (cur = lib_list; cur->next; cur = cur->next) ; /* nothing */
487  cur->next = newlib;
488  }
489  return 0;
490 }

505 static struct library *find_elf_interpreter(ElfW(Ehdr) *ehdr)
506 {
507  ElfW(Phdr) *phdr;
508 
509  if (interpreter_already_found == 1)
510  return NULL;
511  phdr = elf_find_phdr_type(PT_INTERP, ehdr);
512  if (phdr) {
513  struct library *cur, *newlib = NULL;
514  char *s = (char *)ehdr + byteswap_to_host(phdr->p_offset);
515 
516  char *tmp, *tmp1;
517  interp_name = strdup(s);
518  interp_dir = strdup(s);
519  tmp = strrchr(interp_dir, '/');
520  if (tmp)
521  *tmp = '\0';
522  else {
523  free(interp_dir);
524  interp_dir = interp_name;
525  }
526  tmp1 = tmp = s;
527  while (*tmp) {
528  if (*tmp == '/')
529  tmp1 = tmp + 1;
530  tmp++;
531  }
532  for (cur = lib_list; cur; cur = cur->next) {
533  /* Check if this library is already in the list */
534  if (strcmp(cur->name, tmp1) == 0) {
535  /*printf("find_elf_interpreter is replacing '%s' (already in list)\n", cur->name); */
536  newlib = cur;
537  free(newlib->name);
538  if (newlib->path != not_found) {
539  free(newlib->path);
540  }
541  newlib->name = NULL;
542  newlib->path = NULL;
543  break;
544  }
545  }
546  if (newlib == NULL)
547  newlib = malloc(sizeof(struct library));
548  if (!newlib)
549  return NULL;
550  newlib->name = malloc(strlen(s) + 1);
551  strcpy(newlib->name, s);
552  newlib->path = strdup(newlib->name);
553  newlib->resolved = 1;
554  newlib->next = NULL;
555  interpreter_already_found = 1;
556  return newlib;
557  }
558  return NULL;
559 }

561 /* map the .so, and locate interesting pieces */
562 /*
563 #warning "There may be two warnings here about vfork() clobbering, ignore them"
564 */
565 static int find_dependencies(char *filename)
566 {
567  int is_suid = 0;
568  FILE *thefile;
569  struct stat statbuf;
570  ElfW(Ehdr) *ehdr = NULL;
571  ElfW(Shdr) *dynsec = NULL;
572  ElfW(Dyn) *dynamic = NULL;
573  struct library *interp;
574 
575  if (filename == not_found)
576  return 0;
577 
578  if (!filename) {
579  fprintf(stderr, "No filename specified.\n");
580  return -1;
581  }
582  if (!(thefile = fopen(filename, "r"))) {
583  perror(filename);
584  return -1;
585  }
586  if (fstat(fileno(thefile), &statbuf) < 0) {
587  perror(filename);
588  fclose(thefile);
589  return -1;
590  }
591 
592  if ((size_t) statbuf.st_size < sizeof(ElfW(Ehdr)))
593  goto foo;
594 
595  if (!S_ISREG(statbuf.st_mode))
596  goto foo;
597 
598  /* mmap the file to make reading stuff from it effortless */
599  ehdr = mmap(0, statbuf.st_size, PROT_READ | PROT_WRITE, MAP_PRIVATE, fileno(thefile), 0);
600  if (ehdr == MAP_FAILED) {
601  fclose(thefile);
602  fprintf(stderr, "mmap(%s) failed: %s\n", filename, strerror(errno));
603  return -1;
604  }
605 
606 foo:
607  fclose(thefile);
608 
609  /* Check if this looks like a legit ELF file */
610  if (check_elf_header(ehdr)) {
611  fprintf(stderr, "%s: not an ELF file.\n", filename);
612  return -1;
613  }
614  /* Check if this is the right kind of ELF file */
615  if (ehdr->e_type != ET_EXEC && ehdr->e_type != ET_DYN) {
616  fprintf(stderr, "%s: not a dynamic executable\n", filename);
617  return -1;
618  }
619  if (ehdr->e_type == ET_EXEC || ehdr->e_type == ET_DYN) {
620  if (statbuf.st_mode & S_ISUID)
621  is_suid = 1;
622  if ((statbuf.st_mode & (S_ISGID | S_IXGRP)) == (S_ISGID | S_IXGRP))
623  is_suid = 1;
624  /* FIXME */
625  if (is_suid)
626  fprintf(stderr, "%s: is setuid\n", filename);
627  }
628 
629  interpreter_already_found = 0;
630  interp = find_elf_interpreter(ehdr);
631 
632  if (interp
633  && (ehdr->e_type == ET_EXEC || ehdr->e_type == ET_DYN)
634  && ehdr->e_ident[EI_CLASS] == ELFCLASSM
635  && ehdr->e_ident[EI_DATA] == ELFDATAM
636  && ehdr->e_ident[EI_VERSION] == EV_CURRENT
637  && ehdr->e_machine == ELF_ARCH)
638  {
639  struct stat st;
640  if (stat(interp->path, &st) == 0 && S_ISREG(st.st_mode)) {
641  pid_t pid;
642  int status;
